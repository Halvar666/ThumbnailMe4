cmake_minimum_required(VERSION 3.21)

project(ThumbnailMeQt6 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 REQUIRED COMPONENTS Widgets)

if(NOT DEFINED ENV{FFMPEG_ROOT})
    message(FATAL_ERROR "FFMPEG_ROOT environment variable is not set")
endif()

set(FFMPEG_ROOT "$ENV{FFMPEG_ROOT}")
set(FFMPEG_INCLUDE_DIR "${FFMPEG_ROOT}/include")
set(FFMPEG_LIB_DIR "${FFMPEG_ROOT}/lib")

add_executable(ThumbnailMeQt6 WIN32
    main_qt6_shell.cpp
)

target_include_directories(ThumbnailMeQt6 PRIVATE
    "${FFMPEG_INCLUDE_DIR}"
)

target_link_directories(ThumbnailMeQt6 PRIVATE
    "${FFMPEG_LIB_DIR}"
)

target_link_libraries(ThumbnailMeQt6 PRIVATE
    Qt6::Widgets
    avformat
    avcodec
    avutil
    swscale
    swresample
)
